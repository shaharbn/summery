סיכום לגוני מבוא למדעי המחשב

while loop

מיועד למצבים שאינך יודעת כמה פעמים את צריכה לעבור על משהו, זה טוב כדי לעבור על מספרים ואו לעשות משהו כמעט אין סוף פעמים כמו תפריט למשתמש שיכול לבחור דברים

//דברים שאת רוצה לזכור לאחר או בזמן שהלולאה רצה

while(condition) {

// code block to be executed

}

קישור לקריאה במידת הצורך

do while loop

מיועד למצבים שאינך יודעת כמה פעמים את צריכה לעבור על משהו אבל את יודעת שהיא חייבת לרוץ פעם אחת לפחות

do {

// code block to be executed

}

while (condition);

קישור לקריאה במידת הצורך

for loop

טוב למעבר על דברים שאת יודעת כמה פעמים את צריכה לעבור עליהם, למשל מערך

for (int i = 0; condition; i++) {

// code block to be executed

}

קישור לקריאה במידת הצורך

Math

פונקציות מתמטיות שימושיות

Math.max(x,y) - מקסימום בין 2 מספרים

Math.min(x,y) - מינימום בין 2 מספרים

Math.abs(x) - ערך מוחלט זה מקבל מספר יחיד כלומר ניתן גם לשים ביטוי חשבוני

קישור לקריאה במידת הצורך

String

לא לשכוח שמחרוזת בסוף זה מערך של אותיות ומדובר בספרייה ולכן על מחרוזת היא אובייקט שיש עליו מטודות

String txt = "ABC";

String txt = new String("ABC");

txt.toUpperCase() - עושה את כל האותיות גדולות לזכור שזה לא משנה את המשתנה המקורי

txt.toLowerCase() - עושה את כל האותיות קטנות לזכור שזה לא משנה את המשתנה המקורי

txt.indexOf("locate") - מוצא את האינדקס שהמילה הזאתי מתחילה במחרוזת

txt.length() - מחזיר את אורך הטקסט


קישור לקריאה במידת הצורך

מערך חד מימידי

מערך זה בעצם איגוד של נתונים זהים כמו מספרים שלמים אחד ליד השני פיזית בזיכרון

לרוב המעבר על מערך כזה הינו בעזרת לולאת פור יחידה

int []arr = new int[3] - הכרזתי על מערך ריק של מספרים שלמים בגודל שלוש תאים, הגודל בזמן הריצה אינו ניתן לשינוי

int []arr = {10, 20, 30} - הכרזתי פה על מערך בגודל 3 תאים שכבר מכיל לי נתונים

arr[0] - ככה אני פונה לתא במקום הראשון וכן אלעה

הערה חשובה כן ניתן לקבוע את הגודל של המערך בזמן ריצה אבל לא יהיה ניתן לשנות אותו

קישור לקריאה במידת הצורך

מערך דו מימידי

זה בעצם מערך חד מימדי שכל תא שלו מכיל מערך חד מימידי

לרוב מעבר על מערך זה בעזרת לולאת פור מקוננת

int []arr = new int[מסםר שורות][מספר עמודות] - אני מכריז על מערך דו מימדי וגם פה הגודל קבוע

arr[איזה שורה][איזה עמודה] - איך אני פונה לתא מסויים

int[][] arr = { {1, 2, 3, 4}, {5, 6, 7} }; - ככה אני עושה מערך דו מימדי שיש בתוכו כבר איברים ולא ניתן לשנות את גודל המערך

הערה חשובה גם פה ניתן לקבוע את הגודל בזמן ריצה אבל אחרי שקובעים אותו לא ניתן לשנות וכן גם כל מערך בתוך תא יכול להיות בגודל שונה אחד מהשני
